260 private: constexpr void Clamp()
261 {
262 // These comparisons are carefully written to handle NaNs correctly.
263 if (!(this->r >= 0)) { this->r = 0; }
264 if (!(this->g >= 0)) { this->g = 0; }
265 if (!(this->b >= 0)) { this->b = 0; }
266 if (!(this->a >= 0)) { this->a = 0; }
267 if (this->r > 1) { this->r = this->r/255.0f; }
268 if (this->g > 1) { this->g = this->g/255.0f; }
269 if (this->b > 1) { this->b = this->b/255.0f; }
270 if (this->a > 1) { this->a = 1; }
271 }

294 public: friend std::istream &operator>> (std::istream &_in, Color &_pt)
295 {
296 // Skip white spaces
297 _in.setf(std::ios_base::skipws);
298 _in >> _pt.r >> _pt.g >> _pt.b;
299 // Since alpha is optional, check if it's there before parsing
300 while (_in.good() && std::isspace(_in.peek()))
301 {
302 _in.get();
303 }
304 if (_in.good())
305 {
306 _in >> _pt.a;
307 }
308 else if (!_in.fail())
309 {
310 _pt.a = 1.0;
311 }
312 return _in;
313 }
